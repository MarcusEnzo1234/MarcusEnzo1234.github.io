function drawCuteCockatiel(mut){ const w=canvas.width, h=canvas.height; const x=w*0.5, y=h*0.54; const C = mut.draw; // STYLE to match your reference image: thick-ish black outline, simple shading const OUT = "rgba(10,12,18,0.92)"; const OUT_SOFT = "rgba(10,12,18,0.55)"; // --- helper --- function ellipse(cx,cy,rx,ry,rot,fill,strokeW=7,stroke=OUT){ ctx.save(); ctx.translate(cx,cy); ctx.rotate(rot); ctx.beginPath(); ctx.ellipse(0,0,rx,ry,0,0,Math.PI*2); ctx.fillStyle=fill; ctx.fill(); ctx.lineWidth=strokeW; ctx.strokeStyle=stroke; ctx.stroke(); ctx.restore(); } function roundRectFill(x,y,w,h,r,fill,strokeW=6,stroke=OUT){ ctx.save(); ctx.beginPath(); ctx.moveTo(x+r,y); ctx.arcTo(x+w,y,x+w,y+h,r); ctx.arcTo(x+w,y+h,x,y+h,r); ctx.arcTo(x,y+h,x,y,r); ctx.arcTo(x,y,x+w,y,r); ctx.closePath(); ctx.fillStyle=fill; ctx.fill(); ctx.lineWidth=strokeW; ctx.strokeStyle=stroke; ctx.stroke(); ctx.restore(); } // ===== PERCH (like your image) ===== const perchY = y + 255; ctx.save(); ctx.globalAlpha = 0.9; roundRectFill(x-250, perchY, 500, 16, 999, "rgba(120,80,40,0.55)", 5, "rgba(10,12,18,0.65)"); ctx.restore(); // ===== BODY (big fluffy grey blob) ===== // Outer body ellipse(x, y+120, 175, 210, 0, C.body, 8, OUT); // Soft highlight on body (simple) ctx.save(); ctx.globalAlpha=0.18; ellipse(x-40, y+80, 110, 150, -0.15, "#FFFFFF", 0, "transparent"); ctx.restore(); // ===== HEAD (big round yellow) ===== ellipse(x, y-55, 150, 140, 0, C.face, 8, OUT); // ===== CREST (2-3 simple hairs) ===== ctx.save(); ctx.lineCap="round"; ctx.strokeStyle=OUT; ctx.lineWidth=10; // center ctx.beginPath(); ctx.moveTo(x, y-180); ctx.quadraticCurveTo(x-10, y-250, x+15, y-295); ctx.stroke(); // left ctx.beginPath(); ctx.moveTo(x-18, y-175); ctx.quadraticCurveTo(x-55, y-240, x-15, y-285); ctx.stroke(); // right ctx.beginPath(); ctx.moveTo(x+22, y-175); ctx.quadraticCurveTo(x+60, y-235, x+45, y-282); ctx.stroke(); ctx.restore(); // ===== CHEEKS (round blush dots) ===== if(mut.traits.cheeks){ ctx.save(); ctx.globalAlpha=0.95; ctx.fillStyle=C.cheek; ctx.strokeStyle=OUT_SOFT; ctx.lineWidth=4; ctx.beginPath(); ctx.arc(x-78, y-35, 26, 0, Math.PI*2); ctx.fill(); ctx.stroke(); ctx.beginPath(); ctx.arc(x+78, y-35, 26, 0, Math.PI*2); ctx.fill(); ctx.stroke(); ctx.restore(); } // ===== EYES (tiny dots like the reference) ===== ctx.save(); ctx.fillStyle="rgba(10,12,18,0.92)"; ctx.beginPath(); ctx.arc(x-38, y-78, 8, 0, Math.PI*2); ctx.fill(); ctx.beginPath(); ctx.arc(x+38, y-78, 8, 0, Math.PI*2); ctx.fill(); ctx.restore(); // ===== BEAK (tiny triangle-ish) ===== ctx.save(); ctx.fillStyle="rgba(245,196,120,0.95)"; ctx.strokeStyle=OUT_SOFT; ctx.lineWidth=4; ctx.beginPath(); ctx.moveTo(x, y-62); ctx.quadraticCurveTo(x+18, y-52, x, y-38); ctx.quadraticCurveTo(x-18, y-52, x, y-62); ctx.closePath(); ctx.fill(); ctx.stroke(); ctx.restore(); // ===== LITTLE “CHEST FLUFF” line (like the ref) ===== ctx.save(); ctx.strokeStyle=OUT_SOFT; ctx.lineWidth=6; ctx.lineCap="round"; ctx.beginPath(); ctx.moveTo(x-55, y+70); ctx.quadraticCurveTo(x, y+90, x+55, y+70); ctx.stroke(); ctx.restore(); // ===== WINGS (simple dark grey shapes) ===== ctx.save(); ctx.globalAlpha=0.95; ellipse(x-115, y+135, 65, 125, 0.55, C.wing, 7, OUT_SOFT); ellipse(x+115, y+135, 65, 125, -0.55, C.wing, 7, OUT_SOFT); ctx.restore(); // ===== PEARL pattern (subtle) ===== if(mut.traits.pattern==="pearl"){ ctx.save(); ctx.globalAlpha=0.35; ctx.fillStyle="rgba(255,255,255,0.85)"; for(let i=0;i<16;i++){ ctx.beginPath(); ctx.ellipse(x+rand(-105,105), y+rand(40,210), rand(10,16), rand(8,12), rand(0,Math.PI), 0, Math.PI*2); ctx.fill(); } ctx.restore(); } // ===== PIED patches (chunky blocks) ===== if(mut.traits.patches){ ctx.save(); ctx.globalAlpha=0.55; for(let i=0;i<5;i++){ ctx.fillStyle = i%2===0 ? "rgba(155,201,196,0.85)" : "rgba(255,255,255,0.75)"; ctx.beginPath(); ctx.ellipse(x+rand(-90,90), y+rand(40,220), rand(30,55), rand(18,45), rand(-1,1), 0, Math.PI*2); ctx.fill(); } ctx.restore(); } // ===== TAIL (two feathers) ===== ctx.save(); ctx.strokeStyle=OUT_SOFT; ctx.lineWidth=7; ctx.lineCap="round"; // left tail ctx.fillStyle="rgba(70,75,95,0.55)"; ctx.beginPath(); ctx.ellipse(x-18, y+305, 26, 82, 0.05, 0, Math.PI*2); ctx.fill(); ctx.stroke(); // right tail ctx.beginPath(); ctx.ellipse(x+18, y+305, 26, 82, -0.05, 0, Math.PI*2); ctx.fill(); ctx.stroke(); ctx.restore(); // ===== FEET (tiny) on perch ===== ctx.save(); ctx.strokeStyle=OUT_SOFT; ctx.lineWidth=10; ctx.lineCap="round"; ctx.beginPath(); ctx.moveTo(x-55, y+260); ctx.lineTo(x-70, y+290); ctx.stroke(); ctx.beginPath(); ctx.moveTo(x+55, y+260); ctx.lineTo(x+70, y+290); ctx.stroke(); ctx.restore(); }
